45      @PostMapping("/bidList/validate")
46      public String validate(@Valid BidList bid, BindingResult result, Model model, HttpServletResponse response,
47              @RequestParam String bidListDateString, @RequestParam String creationDateString,
48              @RequestParam String revisionDateString) throws ParseException {
49  
50          setDateToFormat(bid, bidListDateString, creationDateString, revisionDateString);
51  
52          if (!result.hasErrors()) {
53              bidListRepository.save(bid);
54              model.addAttribute(BIDS, bidListRepository.findAll());
55              response.setStatus(HttpServletResponse.SC_CREATED); // response 201
56              return REDIRECTBIDLIST;

62      @GetMapping("/bidList/update/{id}")
63      public String showUpdateForm(@PathVariable("id") Integer id, Model model) {
64          BidList bidList = bidListRepository.findById(id)
65                  .orElseThrow(() -> new IllegalArgumentException("Invalid user Id:" + id));
66          model.addAttribute("bidList", bidList);

71      @PostMapping("/bidList/update/{id}")
72      public String updateBid(@PathVariable("id") Integer id, @Valid BidList bidList,
73              BindingResult result, Model model, HttpServletResponse response,
74              @RequestParam String bidListDateString, @RequestParam String creationDateString,
75              @RequestParam String revisionDateString) throws ParseException {
76          if (result.hasErrors()) {
77              response.setStatus(HttpServletResponse.SC_BAD_REQUEST); // response 400
78              return "bidList/update";
79          }
80  
81          setDateToFormat(bidList, bidListDateString, creationDateString, revisionDateString);
82  
83          bidList.setBidlistId(id);
84          bidListRepository.save(bidList);
85          model.addAttribute("bidList", bidListRepository.findAll());
86          response.setStatus(HttpServletResponse.SC_CREATED); // response 201
87          return REDIRECTBIDLIST;
88      }
89  
90      @GetMapping("/bidList/delete/{id}")
91      public String deleteBid(@PathVariable("id") Integer id, Model model) {
92          BidList bidList = bidListRepository.findById(id)
93                  .orElseThrow(() -> new IllegalArgumentException("Invalid user Id:" + id));
94          bidListRepository.delete(bidList);
95          model.addAttribute(BIDS, bidListRepository.findAll());
96          return REDIRECTBIDLIST;
97      }

99      private void setDateToFormat(@Valid BidList bid, String bidListDateString, String creationDateString,
100             String revisionDateString) throws ParseException {
101 
102         if (!bidListDateString.equals("")) {
103             formatDateFormat.setFromatDateStringToTimestamp(bidListDateString);
104             bid.setBidListDate(formatDateFormat.getTimestampFromatDate());
105         }
106         if (!creationDateString.equals("")) {
107             formatDateFormat.setFromatDateStringToTimestamp(creationDateString);
108             bid.setCreationDate(formatDateFormat.getTimestampFromatDate());
109         }
110         if (!revisionDateString.equals("")) {
111             formatDateFormat.setFromatDateStringToTimestamp(revisionDateString);
112             bid.setRevisionDate(formatDateFormat.getTimestampFromatDate());
113         }
114 
115     }
