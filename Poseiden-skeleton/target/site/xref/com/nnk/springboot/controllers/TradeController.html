44              @RequestParam String revisionDateString) throws ParseException {

60          model.addAttribute("trade", trade);

64      @PostMapping("/trade/update/{id}")
65      public String updateTrade(@PathVariable("id") Integer id, @Valid Trade trade,
66              BindingResult result, Model model, @RequestParam String tradeDateString,
67              @RequestParam String creationDateString,
68              @RequestParam String revisionDateString) throws ParseException {
69          if (result.hasErrors()) {
70              return "trade/update";
71          }
72  
73          setDateToFormat(trade, tradeDateString, creationDateString, revisionDateString);
74  
75          trade.setTradeId(id);
76          tradeRepository.save(trade);
77          model.addAttribute("users", tradeRepository.findAll());
78          return "redirect:/trade/list";
79      }
80  
81      @GetMapping("/trade/delete/{id}")
82      public String deleteTrade(@PathVariable("id") Integer id, Model model) {
83          Trade trade = tradeRepository.findById(id)
84                  .orElseThrow(() -> new IllegalArgumentException("Invalid user Id:" + id));
85          tradeRepository.delete(trade);
86          model.addAttribute("trades", tradeRepository.findAll());
87          return "redirect:/trade/list";
88      }

90      private void setDateToFormat(@Valid Trade trade, String tradeDateString, String creationDateString,
91              String revisionDateString) throws ParseException {
92  
93          formatDateFormat.setFromatDateStringToTimestamp(tradeDateString);
94          trade.setTradeDate(formatDateFormat.getTimestampFromatDate());
95          formatDateFormat.setFromatDateStringToTimestamp(creationDateString);
96          trade.setCreationDate(formatDateFormat.getTimestampFromatDate());
