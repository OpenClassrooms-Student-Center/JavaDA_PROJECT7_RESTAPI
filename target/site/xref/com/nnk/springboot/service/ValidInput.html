347     private boolean verifyContraintsPassword(@Valid Users user) {
348 
349         long longPassword = user.getPassword().length();
350         long compteurUpperCase = user.getPassword().chars().filter(Character::isUpperCase).count();
351         long compteurChiffreCase = user.getPassword().chars().filter(Character::isDigit).count();
352         long compteurLetterCase = user.getPassword().chars().filter(Character::isLetter).count();
353 
354         if ((longPassword >= 8) && (compteurUpperCase >= 1) && (compteurChiffreCase >= 1)
355                 && (compteurChiffreCase + compteurLetterCase < longPassword)) {
356 
357             String messageLoggerInfo = loggerApi
358                     .loggerStrings("Le mot de passe a au moins 8 caractères ; ",
359                             "il comporte au moins 1 majuscule et 1 chiffre",
360                             "et il a au moins un symbole (les chiffres + les lettres font moins que la totalité des caractères)");
361 
362             LOGGER.info(messageLoggerInfo);
363             return true;
364 
365         } else {
366 
367             String messageLoggerInfo = loggerApi
368                     .loggerStrings("Le mot de passe ne respecte pas les consignes",
369                             "",
370                             "");
371 
372             LOGGER.info(messageLoggerInfo);
373             return false;
374 
375         }
376     }
